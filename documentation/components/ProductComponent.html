import { Component } from '@angular/core';
import { SharedModule } from '../../../shared/shared.module';
import { Productolist,Producto } from '../interfaces/producto.model';
import { ProductoService } from '../services/producto.service';
import { NzUploadChangeParam } from 'ng-zorro-antd/upload';
import * as XLSX from 'xlsx';
import { NzUploadFile } from 'ng-zorro-antd/upload';
@Component({
  selector: 'app-product',
  standalone: true,
  imports: [SharedModule],
  templateUrl: './product.component.html',
  styleUrl: './product.component.css'
})
export class ProductComponent {
  productos: Productolist[] = [];
  productoLista: Producto[] = [];
  consultaBusqueda: string = '';
  filtroCategoria: string = 'Todos';
  filtroMarca: string = 'Todos';
  filtroEstado: string = 'Todos';
  columnaOrden: string = 'producto_nombre';
  direccionOrden: string = 'asc';
  fileList: any[] = [];
  // Variables para el control del modal
  esVisibleModalAgregarProducto = false;
  nuevoProducto: Partial<Producto> = {};

  constructor(private productoService: ProductoService) {}

  ngOnInit(): void {
    this.obtenerProductos();
  }

  obtenerProductos() {
    this.productoService.obtenerProductos().subscribe(
      (productos: Productolist[]) => {
        this.productos = productos;
      },
      error => {
        console.error('Error al obtener los productos:', error);
      }
    );
  }
handleUploadChange(event: NzUploadChangeParam): void {
  const file = event.file;

  if (file.status === 'done') {
    // Suponiendo que la URL de la imagen esté en `file.response.url`
    this.nuevoProducto.imagen_url = file.response.url;
  } else if (file.status === 'error') {
    console.error('Error al cargar el archivo:', file.error);
  }
}
  productosFiltrados() {
    return this.productos
      .filter(producto => 
        producto.producto_nombre.toLowerCase().includes(this.consultaBusqueda.toLowerCase()) || 
        producto.producto_descripcion.toLowerCase().includes(this.consultaBusqueda.toLowerCase())
      )
      .filter(producto => this.filtroCategoria === 'Todos' || producto.categoria_nombre === this.filtroCategoria)
      .filter(producto => this.filtroMarca === 'Todos' || producto.marca === this.filtroMarca)
      .filter(producto => this.filtroEstado === 'Todos' || 
        (this.filtroEstado === 'Activo' && producto.activo) || 
        (this.filtroEstado === 'Inactivo' && !producto.activo)
      )
      .sort((a, b) => {
        const valorA = a[this.columnaOrden as keyof Productolist]?.toString() ?? '';
        const valorB = b[this.columnaOrden as keyof Productolist]?.toString() ?? '';
        if (valorA < valorB) return this.direccionOrden === 'asc' ? -1 : 1;
        if (valorA > valorB) return this.direccionOrden === 'asc' ? 1 : -1;
        return 0;
      });
  }

  ordenarPor(columna: string) {
    if (columna === this.columnaOrden) {
      this.direccionOrden = this.direccionOrden === 'asc' ? 'desc' : 'asc';
    } else {
      this.columnaOrden = columna;
      this.direccionOrden = 'asc';
    }
  }

  // Métodos para manejar el modal
  abrirModalAgregarProducto() {
    this.esVisibleModalAgregarProducto = true;
    this.nuevoProducto = {}; // Limpiar entrada anterior
  }

  cerrarModalAgregarProducto() {
    this.esVisibleModalAgregarProducto = false;
  }

  agregarProducto() {
    const formData = new FormData();
    formData.append('nombre', this.nuevoProducto.nombre || '');
    formData.append('descripcion', this.nuevoProducto.descripcion || '');
    formData.append('categoria_id', this.nuevoProducto.categoria_id?.toString() || '');
    formData.append('subcategoria_id', this.nuevoProducto.subcategoria_id?.toString() || '');
    formData.append('marca', this.nuevoProducto.marca || '');
    formData.append('precio', this.nuevoProducto.precio?.toString() || '');
    formData.append('cantidad_disponible', this.nuevoProducto.cantidad_disponible?.toString() || '');
    formData.append('unidad_medida', this.nuevoProducto.unidad_medida || '');
    formData.append('activo', this.nuevoProducto.activo ? 'true' : 'false');
    

    // Añadir el archivo de imagen si existe
    if (this.fileList.length > 0) {
        formData.append('imagen_url', this.fileList[0].originFileObj);
    }

    this.productoService.agregarProducto(formData).subscribe(
        (producto: Producto) => {
            this.productoLista.push(producto); // Agregar el nuevo producto a la lista
            this.cerrarModalAgregarProducto();
        },
        error => {
            console.error('Error al agregar el producto:', error);
        }
    );
}

handleExcelUpload = (file: NzUploadFile): boolean => {
  const reader = new FileReader();
  reader.onload = (e: any) => {
    const data = new Uint8Array(e.target.result);
    const workbook = XLSX.read(data, { type: 'array' });
    const firstSheet = workbook.Sheets[workbook.SheetNames[0]];
    const excelData = XLSX.utils.sheet_to_json(firstSheet, { header: 1 }) as any[][];


    this.procesarExcelData(excelData);
  };
  reader.readAsArrayBuffer(file as any);
  return false; // Impide la carga automática del archivo
};

// Procesa el contenido del Excel y lo convierte a una lista de productos
procesarExcelData(data: any[][]): void {
  const headers = data[0];
  const rows = data.slice(1);

  // Mapea cada fila a un objeto producto
  this.productoLista = rows.map(row => {
    return {
      id_producto: 0, // Valor predeterminado o ajusta según sea necesario
      nombre: row[headers.indexOf('Producto')] || 'Producto sin nombre', // Valor predeterminado
      descripcion: 'Descripción por defecto', // Valor predeterminado
      categoria_id: 1, // Ajusta a una categoría válida predeterminada
      subcategoria_id: 1, // Valor predeterminado para subcategoría
      marca: row[headers.indexOf('Marca')] || 'Marca genérica', // Valor predeterminado
      precio: Number(row[headers.indexOf('Precio')]) || 1, // Precio predeterminado para evitar 0 o ''
      cantidad_disponible: 1, // Valor predeterminado para cantidad disponible
      unidad_medida: 'unidad', // Valor predeterminado (p.ej., 'unidad', 'kg', etc.)
      imagen_url: null, // No se usa en el archivo Excel
      fecha_agregado: new Date(), // Fecha actual
      activo: true // Valor predeterminado
    } as Producto;
  });
  
  
  

  console.log(this.productos); // Aquí puedes revisar los productos procesados
}


  editarProducto(producto: Producto) {
    // Agregar funcionalidad de edición aquí
  }
}
