@Component({
  selector: 'app-admin-sidebar',
  standalone: true,
  imports: [SharedModule,NzMenuModule],
  schemas: [CUSTOM_ELEMENTS_SCHEMA],
  templateUrl: './admin-sidebar.component.html',
  styleUrl: './admin-sidebar.component.css'
})
export class AdminSidebarComponent {

  activeTab: string = 'overview';

  // Lista de pestañas
  tabs: string[] = [
    'analytics',
    'products',
    'categories',
    'customers',
    'orders',
    'reports'
  ];

  constructor(private router: Router) {}

  // Método para actualizar la pestaña activa
  setActiveTab(tab: string): void {
    this.activeTab = tab;
  }

  // Define el tipo de ícono según la pestaña
  getIconType(tab: string): string {
    switch (tab) {
      case 'analytics': return 'pie-chart';
      case 'products': return 'appstore';
      case 'categories': return 'folder';
      case 'customers': return 'user-add';
      case 'orders': return 'shopping-cart';
      case 'reports': return 'file-text';
      default: return 'file';
    }
  }

  // Define la ruta de cada pestaña
  getRoute(tab: string): string {
    switch (tab) {
      case 'analytics': return '/admin/dashboard';
      case 'products': return '/admin/product';
      case 'categories': return '/admin/categorie';
      case 'customers': return '/admin/customer';
      case 'orders': return '/admin/order';
      case 'reports': return '/admin/report';
      default: return '/admin/dashboard';
    }
  }



  isCollapsed = false;
   // Información del drawer (sidebar)
  drawerInfo = {
    content: {
      menu: true,
      menuHeader: true
    }
  };

  // Estilo del sitio, restringido a 'light' o 'dark'
  siteStyle: 'light' | 'dark' = 'light'; // Puedes cambiar a 'dark' según tu lógica

  // Menú de ejemplo
  menuList: MenuItem[] = [
