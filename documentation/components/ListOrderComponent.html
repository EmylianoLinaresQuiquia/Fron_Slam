import { Component } from '@angular/core';
import { SharedModule } from '../../../shared/shared.module';
import { Router } from '@angular/router';
@Component({
  selector: 'app-list-order',
  standalone: true,
  imports: [SharedModule],
  templateUrl: './list-order.component.html',
  styleUrl: './list-order.component.css'
})
export class ListOrderComponent {
  orders = [
    { orderNumber: '001', customerName: 'Juan Pérez', date: new Date(), status: 'Completado', total: 250.00 },
    { orderNumber: '002', customerName: 'Ana López', date: new Date(), status: 'Pendiente', total: 150.00 }
  ];

  constructor(private router: Router) {}

  viewOrderDetails(order: any) {
    // Navegar al detalle del pedido
    this.router.navigate(['/order-detail', order.orderNumber]);
  }
}
