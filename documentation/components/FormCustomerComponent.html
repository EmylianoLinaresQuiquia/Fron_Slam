import { Component } from '@angular/core';
import { SharedModule } from '../../../shared/shared.module';
import { ClienteService } from '../services/cliente.service';
import { Cliente } from '../interfaces/cliente.model';
@Component({
  selector: 'app-form-customer',
  standalone: true,
  imports: [SharedModule],
  templateUrl: './form-customer.component.html',
  styleUrl: './form-customer.component.css'
})
export class FormCustomerComponent {
  newCliente: Cliente = {
    idCliente: 0, // Si es auto-incremental o lo manejas en el backend
    nombre: '',
    email: '',
    telefono: '',
    direccionCalle: '', // Rellena según sea necesario
    direccionCiudad: '',
    direccionCodigoPostal: '',
    direccionPais: '',
    fechaRegistro: new Date(), // O puedes manejar esto en el backend
    activo: true // Si deseas que el cliente esté activo por defecto
  };

  constructor(private clienteService: ClienteService) {} // Inyectar el servicio

  onSubmit() {
    // Lógica para agregar el nuevo cliente
    this.clienteService.agregarCliente(this.newCliente).subscribe({
      next: (cliente) => {
        console.log('Cliente agregado:', cliente);
        // Reiniciar el formulario
        this.resetForm();
      },
      error: (err) => {
        console.error('Error al agregar cliente:', err);
        // Manejar error según sea necesario
      }
    });
  }

  resetForm() {
    this.newCliente = {
      idCliente: 0,
      nombre: '',
      email: '',
      telefono: '',
      direccionCalle: '',
      direccionCiudad: '',
      direccionCodigoPostal: '',
      direccionPais: '',
      fechaRegistro: new Date(),
      activo: true
    };
  }
}
