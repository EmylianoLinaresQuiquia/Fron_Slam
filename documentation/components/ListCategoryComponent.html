import { Component } from '@angular/core';
import { SharedModule } from '../../../shared/shared.module';
import { CategoriaService } from '../services/categoria.service';
import { Categoria } from '../interfaces/categoria.model';
@Component({
  selector: 'app-list-category',
  standalone: true,
  imports: [SharedModule],
  templateUrl: './list-category.component.html',
  styleUrl: './list-category.component.css'
})
export class ListCategoryComponent {
  categories: Categoria[] = [];

  constructor(private categoriaService: CategoriaService) {}

  ngOnInit() {
    this.loadCategories();
  }

  loadCategories() {
    this.categoriaService.obtenerCategorias().subscribe(
      (categorias) => {
        console.log("data categoria",categorias),
        this.categories = categorias;
      },
      (error) => {
        console.error('Error al cargar las categorías:', error);
      }
    );
  }

  editCategory(category: Categoria) {
    // Lógica para editar la categoría (puedes implementar el detalle de cómo se edita)
    console.log('Editando categoría:', category);
  }

  deleteCategory(category: Categoria) {
    this.categories = this.categories.filter(c => c !== category);
    console.log('Categoría eliminada:', category);
  }
}
