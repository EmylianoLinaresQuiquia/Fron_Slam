import { Component } from '@angular/core';
import { Router } from '@angular/router';
import { SharedModule } from '../../../../shared/shared.module';
import { NzDropDownModule,NzPlacementType  } from 'ng-zorro-antd/dropdown';
@Component({
  selector: 'app-sales-header',
  standalone: true,
  imports: [SharedModule,NzDropDownModule],
  templateUrl: './sales-header.component.html',
  styleUrl: './sales-header.component.css'
})
export class SalesHeaderComponent {
  vegetables = [
    { name: 'pepper', count: 3 },
    { name: 'carrot', count: 8 },
    { name: 'onion', count: 2 },
    { name: 'lettuce', count: 1 },
    { name: 'tomato', count: 2 }
  ];

  additionalItems = ['Cucumber', 'Brinjal', 'Cabbage', 'Beetroot'];

  sizes = [
    { size: 'S', weight: '3 kg' },
    { size: 'M', weight: '4 kg' },
    { size: 'L', weight: '5 kg' },
    { size: 'XL', weight: '6 kg' }
  ];

  packages = ['Bag', 'Basket', 'Box'];

  selectedSize = 'L';
  selectedPackage = 'Basket';
  selectedCity = '';
  doorDelivery = false;
  totalPrice = 32.45;

  incrementCount(veg: any) {
    veg.count++;
  }

  decrementCount(veg: any) {
    if (veg.count > 0) {
      veg.count--;
    }
  }

  addToCart() {
    console.log('Added to cart');
  }
}
