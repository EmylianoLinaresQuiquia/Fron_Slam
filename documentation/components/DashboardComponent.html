import { Component,ElementRef,ViewChild,AfterViewInit } from '@angular/core';
import { SharedModule } from '../../../shared/shared.module';
import { Chart } from '@antv/g2';
import { CurrencyPipe } from '@angular/common';
import { CdkDragDrop, moveItemInArray } from '@angular/cdk/drag-drop';
@Component({
  selector: 'app-dashboard',
  standalone: true,
  imports: [SharedModule],
  providers: [CurrencyPipe],
  templateUrl: './dashboard.component.html',
  styleUrl: './dashboard.component.css'
})
export class DashboardComponent implements AfterViewInit {
  @ViewChild('salesChart', { static: true }) salesChartContainer!: ElementRef;
  @ViewChild('categoryChart', { static: true }) categoryChartContainer!: ElementRef;
  @ViewChild('topProductsChart', { static: true }) topProductsChartContainer!: ElementRef;
  @ViewChild('orderStatusChart', { static: true }) orderStatusChartContainer!: ElementRef;
  totalStock = 320;
  monthlyRevenue = 15000;

  constructor(private currencyPipe: CurrencyPipe) {}

  topProducts = [
    { name: 'Laptop', sales: 120 },
    { name: 'Smartphone', sales: 90 },
    { name: 'Audífonos', sales: 70 }
  ];

  salesData = [
    { day: 'Lun', sales: 500 },
    { day: 'Mar', sales: 700 },
    { day: 'Mié', sales: 800 },
    { day: 'Jue', sales: 600 },
    { day: 'Vie', sales: 900 },
    { day: 'Sáb', sales: 1000 },
    { day: 'Dom', sales: 1200 }
  ];

  categoryData = [
    { category: 'Electrónica', value: 45 },
    { category: 'Ropa', value: 25 },
    { category: 'Hogar', value: 20 },
    { category: 'Accesorios', value: 10 }
  ];

  productList = [
    {
      nombre: 'Pasta con Pesto',
      categoria: 'Alimentos',
      subcategoria: 'Pastas',
      marca: 'Marca X',
      precio: 5.99,
      cantidad_disponible: 100,
      activo: true
    },
    {
      nombre: 'Salsa de Tomate',
      categoria: 'Alimentos',
      subcategoria: 'Salsas',
      marca: 'Marca Y',
      precio: 3.50,
      cantidad_disponible: 50,
      activo: true
    }
  ];

  orderSummary = [
    { status: 'Pendiente', count: 8 },
    { status: 'Enviado', count: 5 },
    { status: 'Entregado', count: 20 },
    { status: 'Cancelado', count: 2 }
  ];

  ngAfterViewInit() {
    this.createSalesChart();
    this.createCategoryChart();
    this.createTopProductsChart();
    this.createOrderStatusChart();
  }


  // Gráfico de Ventas Recientes
  createSalesChart() {
    const chart = new Chart({
      container: this.salesChartContainer.nativeElement,
      autoFit: true,
      height: 400,
    });
    chart.data(this.salesData);
    chart.scale({ sales: { min: 0 } });
    chart.line()
      .encode('x', 'day')
      .encode('y', 'sales')
      .encode('color', '#4caf50')
      .encode('shape', 'smooth');
    chart.render();
  }

  // Gráfico de Barras para Distribución de Categorías
  createCategoryChart() {
    const chart = new Chart({
      container: this.categoryChartContainer.nativeElement,
      autoFit: true,
      height: 400,
    });
    chart.data(this.categoryData);
  
    chart.interval()
      .encode('x', 'category')
      .encode('y', 'value')
      .encode('color', 'category')
      .label({
        fields: ['value'], // Define el campo de la etiqueta
        style: {
          fill: '#000',
          fontSize: 12,
        },
      });
  
    chart.axis('category', { title: { text: 'Categorías' } });
    chart.axis('value', { title: { text: 'Valores' } });
    chart.render();
  }
  
  

  // Gráfico de Barras para Top Productos Vendidos
  createTopProductsChart() {
    const chart = new Chart({
      container: this.topProductsChartContainer.nativeElement,
      autoFit: true,
      height: 400,
    });
    chart.data(this.topProducts);
  
    chart.interval()
      .encode('x', 'name')
      .encode('y', 'sales')
      .encode('color', 'name')
      .label({
        fields: ['sales'], // Define el campo de la etiqueta
        style: {
          fill: '#000',
          fontSize: 12,
        },
      });
  
    chart.axis('name', { title: { text: 'Productos' } });
    chart.axis('sales', { title: { text: 'Ventas' } });
    chart.render();
  }
  
  

  // Gráfico de Líneas para Estado de Pedidos
  createOrderStatusChart() {
    const chart = new Chart({
      container: this.orderStatusChartContainer.nativeElement,
      autoFit: true,
      height: 400,
    });
    chart.data(this.orderSummary);
    chart.line()
      .encode('x', 'status')
      .encode('y', 'count')
      .encode('color', 'status');
      
    chart.point()
      .encode('x', 'status')
      .encode('y', 'count')
      .style({ fill: 'white', stroke: '#5B8FF9', lineWidth: 2 });

    chart.axis('status', { title: { text: 'Estado de Pedido' } });
    chart.axis('count', { title: { text: 'Cantidad' } });
    chart.render();
  }

  // Formateo de Ingresos Mensuales
  get formattedMonthlyRevenue() {
    return this.currencyPipe.transform(this.monthlyRevenue, 'USD');
  }
}
