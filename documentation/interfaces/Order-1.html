import { Component } from '@angular/core';
import { SharedModule } from '../../../shared/shared.module';
import { NzDropDownModule } from 'ng-zorro-antd/dropdown';

import { NzMenuModule } from 'ng-zorro-antd/menu';

export interface Order {
  id_pedido: string;
  cliente: Cliente;
  productos: Producto[];
  total: number;
  fecha_pedido: string;
  estado: EstadoPedido;
  direccion_entrega: DireccionEntrega;
}

export interface Cliente {
  nombre: string;
  email: string;
}

export interface Producto {
  nombre: string;
  cantidad: number;
  precio_unitario: number;
}

export interface DireccionEntrega {
  calle: string;
  ciudad: string;
  codigo_postal: string;
  pais: string;
}

export type EstadoPedido = 'pendiente' | 'enviado' | 'entregado' | 'cancelado';
@Component({
  selector: 'app-order',
  standalone: true,
  imports: [SharedModule,NzDropDownModule,NzMenuModule],
  templateUrl: './order.component.html',
  styleUrl: './order.component.css'
})
export class OrderComponent {
  loadingAll: boolean = false;
  orders = [
    { id: 'ORD123456', cliente: 'Juan Pérez', total: 15.48, fecha: new Date(), estado: 'Pendiente' },
    // Otros pedidos
  ];
  filteredOrders = this.orders;

  filterByStatus(status: string): void {
    this.filteredOrders = status ? this.orders.filter(order => order.estado === status) : this.orders;
  }

  getStatusColor(status: string): string {
    switch (status) {
      case 'Pendiente':
        return 'gold';
      case 'Enviado':
        return 'blue';
      case 'Entregado':
        return 'green';
      case 'Cancelado':
        return 'red';
      default:
        return 'default';
    }
  }

  viewOrderDetail(order:any): void {
    console.log('Viewing details for:', order);
    // Implementa la funcionalidad de detalle de pedido
  }

}
